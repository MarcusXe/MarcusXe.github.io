MIPS32

2021-01-21 HardWare 780 字
Archtacture

16位立即数拓展为32位立即数

符号拓展即将高位全部置1,无符号拓展即将高位全部置0

16位立即数 0X8000 0X1000
符号拓展 0XFFFF8000 0XFFFF1000
无符号拓展 0X00008000 0X00001000

MIPS32指令格式:

  • R型
op(6bit) rs(5bit) rt(5bit) rd(5bit) sa(5bit) func(6bit)
操作码 源寄存器1 源寄存器2 目标寄存器 移位位数 运算码
  • I型
op(6bit) rs(5bit) rt(5bit) immediate(16bit)
操作码 源寄存器1 源寄存器2 立即数
  • J型
op(6bit) address
操作码 字地址

指令寄存器

pc_reg.v

`include "defines.v"
module pc_reg(
	input	wire				clk,	//Clock
	input	wire				rst,	//reset
	output	reg[`InstAddrBus] 	pc,		//inst addr
	output	reg 				ce,		//inst mem enable
);
	always @(posedge clk) begin
		if(rst==`RstEnable) begin
			ce<=`ChipDisable;
		end else begin
			ce<=`ChipEnable;
		end
	end

	always @(posedge clk) begin
		if (ce==`ChipDisable) begin
			pc<=`ZeroWord;			
		end
		else begin
			pc<=pc+4'h4;//Byte addressing
		end
	end
endmodule

指令暂存,在下一个时钟周期进行译码

if_id.v

//fetch instruction to decode instruction 
`incude "defines.v"
module if_id(
	input wire 					clk,
	input wire 					rst,
    input wire [`InstAddrBus]	if_pc,//inst addr
    input wire [`InstBus]		if_inst,//inst from rom..i guess
	output reg	[`InstAddrBus]	id_pc,
	output reg	[`InstBus]		id_inst
);
	always @(posedge clk)begin
		if (rst==`RstEnable) begin
			id_pc<=`ZeroWord;
			id_inst<=`ZeroWord;
		end else begin
			id_pc<=if_pc;
			id_inst<=if_inst;
		end
	end
endmodule

寄存器堆,mips规定0号寄存器不给用

regfile.v

`include "defines.v"
module regfile(
	input wire clk,
	input wire rst,
	//Write port
	input wire 		we,//write enable
	input wire[`RegAddrBus] 	waddr,
	input wire[`RegBus]			wdata,
	//read port 1
	input wire 		re1,//read 1 enable
	input wire[`RegAddrBus]		raddr1,
	input wire[`RegBus]			rdata1,
	//read port 2
	input wire 		re2,
	input wire [`RegAddrBus]	raddr2,
	input wire [`RegBus]		rdata2		
);
reg[`RegBus]regs[0:`RegNum-1];

//write
	always @(posedge clk) begin
		if (rst==`RstDisable) begin
			if((we==`WriteEnable)&&(waddr!=`RegNumLog2'h0)) begin
				regs[waddr]<=wdata;
			end		
		end
	end
//read 1
	always @(*) begin
		if (rst==`RstEnable) begin
			rdata1<=`ZeroWord;
		end
		else if (radddr1==`RegNumLog2'h0) begin
			rdata1<=`ZeroWord;
		end else if((radddr1==waddr)&&(we==`WriteEnable)
						&&(re1==`ReadEnable)) begin
			rdata1<=wdata;
		end else if(re1==`ReadEnable)begin
			rdata1<=regs[raddr1];
		end else begin
			rdata1<=`ZeroWord;
		end
	end
 //read 2
	always @(*) begin
		if (rst==`RstEnable) begin
			rdata2<=`ZeroWord;
		end
		else if (radddr2==`RegNumLog2'h0) begin
			rdata2<=`ZeroWord;
		end else if((radddr2==waddr)&&(we==`WriteEnable)
						&&(re2==`ReadEnable)) begin
			rdata2<=wdata;
		end else if(re2==`ReadEnable)begin
			rdata2<=regs[raddr2];
		end else begin
			rdata2<=`ZeroWord;
		end
	end
endmodule
